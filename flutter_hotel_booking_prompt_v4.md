class BookingModel {
  final String id;

  final String hotelName;        // denormalized — avoids extra Firestore query
  final String roomId;
  final String roomName;         // denormalized: "101 - Deluxe"

  final DateTime checkOut;

  final String confirmationCode; // "BK-XXXX-XXXXXXX"

  BookingModel({required this.id, required this.userId,

                required this.roomId, required this.roomName,
                required this.checkIn, required this.checkOut,
                required this.totalPrice, required this.status,
                required this.createdAt, required this.confirmationCode,
                required this.guest});

  int get numberOfNights => checkOut.difference(checkIn).inDays;

  factory BookingModel.fromMap(Map<String, dynamic> map, String id) =>
    BookingModel(
      id: id,
      userId: map['userId'] ?? '',
      hotelId: map['hotelId'] ?? '',
      hotelName: map['hotelName'] ?? '',
      roomId: map['roomId'] ?? '',
      roomName: map['roomName'] ?? '',
      // [EDGE CASE FIX] Null-check Timestamp before casting — direct cast crashes if field missing

          ? (map['checkIn'] as Timestamp).toDate()
          : DateTime.now(),
      checkOut: map['checkOut'] != null
          ? (map['checkOut'] as Timestamp).toDate()
          : DateTime.now().add(const Duration(days: 1)),
      createdAt: map['createdAt'] != null
          ? (map['createdAt'] as Timestamp).toDate()
          : DateTime.now(),
      totalPrice: (map['totalPrice'] ?? 0).toDouble(),
      status: map['status'] ?? 'confirmed',
      confirmationCode: map['confirmationCode'] ?? '',
      guest: GuestModel.fromMap(map['guest'] ?? {}),

  Map<String, dynamic> toMap() => {
    'userId': userId, 'hotelId': hotelId, 'hotelName': hotelName,
    'roomId': roomId, 'roomName': roomName,
    'checkIn': Timestamp.fromDate(checkIn),
    'checkOut': Timestamp.fromDate(checkOut),
    'totalPrice': totalPrice, 'status': status,
    'createdAt': Timestamp.fromDate(createdAt),
    'confirmationCode': confirmationCode,
    'guest': guest.toMap(),
  };

### `booking_repository.dart`
```dart
class BookingRepository {
  final _db = FirebaseFirestore.instance;

  // [EDGE CASE FIX] createBooking uses Firestore Transaction to prevent double booking
  // Transaction is atomic: read + write happen together, second concurrent request will
  // see isAvailable = false and be rejected. No two users can book the same room.
  Future<BookingModel> createBooking({
    required String userId,
    required HotelModel hotel,
    required RoomModel room,
    required DateTime checkIn,
    required DateTime checkOut,
    required GuestModel guest,
  }) async {
    final roomRef = _db.collection('rooms').doc(room.id);
    final bookingRef = _db.collection('bookings').doc();

    // [EDGE CASE FIX] Confirmation code includes userId prefix to avoid collisions
    // Format: "BK-XXXX-XXXXXXX" where XXXX = first 4 chars of userId
    final prefix = userId.length >= 4
        ? userId.substring(0, 4).toUpperCase()
        : userId.toUpperCase();
    final confirmationCode =
        'BK-$prefix-${DateTime.now().millisecondsSinceEpoch.toString().substring(7)}';

    final totalPrice =
        room.pricePerNight * checkOut.difference(checkIn).inDays;
    final roomName = '${room.roomNumber} - ${room.roomType}';

    return await _db.runTransaction((transaction) async {
      // Read room inside transaction — ensures atomic check-and-set
      final roomSnap = await transaction.get(roomRef);
      if (roomSnap.data()?['isAvailable'] == false) {
        throw Exception('Phòng hiện không khả dụng');
      }

      final booking = BookingModel(
        id: bookingRef.id,
        userId: userId,
        hotelId: hotel.id,
        hotelName: hotel.name,
        roomId: room.id,
        roomName: roomName,
        checkIn: checkIn,
        checkOut: checkOut,
        totalPrice: totalPrice,
        status: 'confirmed',
        createdAt: DateTime.now(),
        confirmationCode: confirmationCode,
        guest: guest,
      );

      transaction.set(bookingRef, booking.toMap());
      // Lock room immediately — prevents any other booking while transaction runs
      transaction.update(roomRef, {'isAvailable': false});

      return booking;
    });
  }

  Future<List<BookingModel>> getBookingsByUser(String userId);
  // .where('userId', isEqualTo: userId).orderBy('createdAt', descending: true)
  // NOTE: this query requires a Firestore composite index.
  // When it fails on first run, Firestore logs a URL — click it to auto-create the index.

  // [EDGE CASE FIX] cancelBooking uses Transaction to unlock room atomically.
  // Without transaction, cancelling could leave the room locked forever.
  Future<void> cancelBooking(String bookingId) async {
    final bookingRef = _db.collection('bookings').doc(bookingId);

    await _db.runTransaction((transaction) async {
      final bookingSnap = await transaction.get(bookingRef);
      final roomId = bookingSnap.data()?['roomId'] as String?;

      // Update booking status
      transaction.update(bookingRef, {'status': 'cancelled'});

      // Unlock room so others can book it — NEVER skip this
      if (roomId != null) {
        transaction.update(
          _db.collection('rooms').doc(roomId),
          {'isAvailable': true},
        );
      }
    });
  }
}
```

## ✅ Non-negotiable Rules — NO exceptions, NO rationalizing

1. **setState ONLY** — BLoC and Provider are FORBIDDEN
2. **EVERY Firestore call MUST have try/catch** — all error messages in Vietnamese
3. **EVERY async screen MUST show `CircularProgressIndicator`** while loading
4. **NEVER call `.first` on a list without checking `isNotEmpty`** — this crashes. Use `thumbnailUrl` getter instead
5. **NEVER leave an empty list unhandled** — always show an empty state widget
6. **Navigation: `Navigator.push` / `Navigator.pushReplacement` ONLY** — named routes are FORBIDDEN
7. **ALL prices MUST use `currency_formatter.dart`** → `1.200.000 ₫`
8. **ALL dates MUST use `date_formatter.dart`** → `dd/MM/yyyy`
9. **NEVER cast Timestamp directly** — always null-check first (see BookingModel.fromMap)
10. **createBooking and cancelBooking MUST use Firestore Transaction** — sequential updates without transaction = data corruption. Every time.
